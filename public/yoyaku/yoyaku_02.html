In order to display the time intervals in a sorted manner, add the
:order parameter to the controller:


yoyaku/app/controllers/course_times_controller.rb

class CourseTimesController < ApplicationController
  # GET /course_times
  # GET /course_times.xml
  def index
    @course_times = CourseTime.find(:all, :order=>'text' )

    respond_to do |format|
      format.html # index.html.erb
      format.xml  { render :xml => @course_times }
    end
  end

  ...

end
